This is the id card logic how it is generated in the firebase/firestire db 

"use client";

import React from "react";
import { User, Student } from "@/lib/definitions";
import Image from "next/image";
import { Card, CardContent } from "@/components/ui/card";
import { Avatar, AvatarFallback, AvatarImage } from "@/components/ui/avatar";
import { User as UserIcon } from "lucide-react";

interface IdCardGeneratorProps {
  user: User;
  idCardRef: React.RefObject<HTMLDivElement>;
}

function isStudent(user: User): user is Student {
  return 'rollno' in user;
}

const IdCardGenerator = React.forwardRef<HTMLDivElement, { user: User }>(({ user }, ref) => {
  const idValue = isStudent(user) ? user.rollno : user.facultyId;
  const detailValue = isStudent(user) ? `${user.branch} - ${user.year}` : user.department;
  const barcodeUrl = user.barcodeImageUrl;

  return (
    <Card ref={ref} className="w-full max-w-sm mx-auto font-sans border-2 border-primary rounded-2xl shadow-lg overflow-hidden bg-background">
      <CardContent className="p-0">
        <div className="bg-primary text-primary-foreground text-center py-3">
          <h2 className="text-xl font-bold tracking-wider">
            INSTITUTION NAME
          </h2>
        </div>
        <div className="p-4 flex flex-col items-center">
          <Avatar className="w-24 h-24 border-4 border-accent mb-3 rounded-md">
            <AvatarImage src={user.photoUrl} alt={user.name} />
            <AvatarFallback className="rounded-md">
              <UserIcon className="h-12 w-12" />
            </AvatarFallback>
          </Avatar>
          <h3 className="text-2xl font-bold text-foreground">{user.name}</h3>
          <p className="text-md text-muted-foreground">{isStudent(user) ? 'Student' : 'Faculty'}</p>
        </div>
        <div className="bg-muted px-4 py-3">
          <div className="flex justify-between items-center mb-2">
            <span className="text-sm font-semibold text-foreground">
              {isStudent(user) ? 'Roll No:' : 'Faculty ID:'}
            </span>
            <span className="text-sm font-mono bg-background px-2 py-0.5 rounded">
              {idValue}
            </span>
          </div>
          <div className="flex justify-between items-center">
            <span className="text-sm font-semibold text-foreground">
              {isStudent(user) ? 'Branch:' : 'Department:'}
            </span>
            <span className="text-sm">
              {detailValue}
            </span>
          </div>
        </div>
        <div className="p-4 flex justify-center items-center flex-col">
          {barcodeUrl && (
            <Image
              src={barcodeUrl}
              alt="Barcode"
              width={250}
              height={50}
              className="object-contain"
            />
          )}
        </div>
      </CardContent>
    </Card>
  );
});

IdCardGenerator.displayName = 'IdCardGenerator';

export default IdCardGenerator;
